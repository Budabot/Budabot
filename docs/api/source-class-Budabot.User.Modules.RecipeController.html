 64:      * @HandlesCommand("recipe")
 65:      * @Matches("/^recipe (.+)$/i")
 66:      */
 67:     public function recipeSearchCommand($message, $channel, $sender, $sendto, $args) {
 68:         if (preg_match('|<a href="itemref://(\d+)/(\d+)/(\d+)">([^<]+)</a>|', $args[1], $matches)) {
 69:             $lowId = $matches[1];
 70:             $search = $matches[4];
 71:             
 72:             $data = $this->db->query("SELECT * FROM recipes WHERE recipe LIKE ? OR recipe LIKE ? ORDER BY name ASC", $search, "%" . $lowId . "%");
 73:         } else {
 74:             $search = $args[1];
 75:             
 76:             list($query, $queryParams) = $this->util->generateQueryFromParams(explode(" ", $search), "recipe");
 77:             $data = $this->db->query("SELECT * FROM recipes WHERE $query ORDER BY name ASC", $queryParams);
 78:         }
 79:         
 80:         $count = count($data);
 81: 
 82:         if ($count == 0) {
 83:             $msg = "Could not find any recipes matching your search criteria.";
 84:         } else if ($count == 1) {
 85:             $msg = $this->createRecipeBlob($data[0]);
 86:         } else {
 87:             $blob = '';
 88:             forEach ($data as $row) {
 89:                 $blob .= $this->text->makeChatcmd($row->name, "/tell <myname> recipe $row->id") . "\n";
 90:             }
 91: 
 92:             $msg = $this->text->makeBlob("Recipes matching '$search' ($count)", $blob);
 93:         }
 94: 
 95:         $sendto->reply($msg);
 96:     }

 98:     public function formatRecipeText($input) {
 99:         $input = str_replace("\\n", "\n", $input);
100:         $input = preg_replace_callback('/#L "([^"]+)" "([0-9]+)"/', array($this, 'replaceItem'), $input);
101:         $input = preg_replace('/#L "([^"]+)" "([^"]+)"/', "<a href='chatcmd://\\2'>\\1</a>", $input);
102:         
103:         // we can't use <myname> in the sql since that will get converted on load,
104:         // and we need to wait to convert until display time due to the possibility
105:         // of several bots sharing the same db
106:         $input = str_replace("{myname}", "<myname>", $input);
107: 
108:         $input = str_replace("#C09","</font><font color=#FFFFFF>", $input);
109:         $input = str_replace("#C12","</font><font color=#FF0000>", $input);
110:         $input = str_replace("#C13","</font><font color=#FFFFFF>", $input);
111:         $input = str_replace("#C14","</font><font color=#FFFFFF>", $input);
112:         $input = str_replace("#C15","</font><font color=#FFFFFF>", $input);
113:         $input = str_replace("#C16","</font><font color=#FFFF00>", $input);
114:         $input = str_replace("#C18","</font><font color=#AAFF00>", $input);
115:         $input = str_replace("#C20","</font><font color=#009B00>", $input);
116:             
117:         return $input;
118:     }

120:     public function createRecipeBlob($row) {
121:         $recipe_name = $row->name;
122:         $author = empty($row->author) ? "Unknown" : $row->author;
123: 
124:         $recipeText = "Author: <highlight>$author<end>\n\n";
125:         $recipeText .= $this->formatRecipeText($row->recipe);
126: 
127:         return $this->text->makeBlob("Recipe for $recipe_name", $recipeText);
128:     }

130:     private function replaceItem($arr) {
131:         $id = $arr[2];
132:         $row = $this->itemsController->findById($id);
133:         if ($row !== null) {
134:             $output = $this->text->makeItem($row->lowid, $row->highid, $row->highql, $row->name);
135:         } else {
136:             $obj = $this->itemsController->getDetailedItemInfo($id);
137:             if (null == $obj) {
138:                 $output = "#L \"{$arr[1]}\" \"/tell <myname> itemid {$arr[2]}\"";
139:             } else if ($obj->icon == 0) {  // for perks and items that aren't displayable in game
140:                 $output = $this->text->makeChatcmd($obj->name, "/start https://aoitems.com/item/{$obj->lowid}");
141:             } else {
142:                 $output = $this->text->makeItem($obj->lowid, $obj->highid, $obj->highql, $obj->name);
143:             }
144:         }
145:         return $output;
